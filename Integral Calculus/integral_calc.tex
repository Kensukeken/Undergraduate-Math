\usepackage{titlesec}

% Define a new counter for the custom section numbering
\newcounter{customsec}
\renewcommand{\thecustomsec}{209\Alph{part}.\arabic{customsec}}

% Command to reset and increment the custom section counter
\newcommand{\resetcustomsec}{
    \setcounter{customsec}{0}
}
\newcommand{\stepcustomsec}{
    \stepcounter{customsec}
    \thesection
}

% Redefine the section command to use the custom numbering
\titleformat{\section}{hang}
  {\normalfont\Large\bfseries}{\thesection}{5em}{}
\renewcommand{\thesection}{\thecustomsec}

% Hook into the part command to reset the custom section counter
\makeatletter
\@addtoreset{section}{part}
\@addtoreset{customsec}{part}
\makeatother

% Increase the section counter whenever a new section is added
\titleformat{\section}[block]
  {\Large\bfseries}
  {\stepcustomsec}{1.5em}{}
  \begin{document}
  hi
  \end{document}
